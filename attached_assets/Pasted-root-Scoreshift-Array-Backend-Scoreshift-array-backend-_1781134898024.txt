  if (!auth || auth !== `Bearer ${process.env.INTERNAL_API_SECRET}`) {
    return res.status(401).json({ error: "Unauthorized" });
  }
  const { userId, productCode } = req.body;

  }
  const creds = Buffer.from(

      "Content-Type": "application/json",
    },
    body: JSON.stringify({ userId, productCode }),
  })

      const body = await r.json().catch(() => ({}));
      console.log(`[Array] enroll ${r.status} userId=${userId} product=${productCode}`);
      res.status(r.status).json(body);
    })
    .catch((e) => {
      console.error("[Array] enroll error:", e.message);
      res.status(502).json({ error: e.message });
    });
});
